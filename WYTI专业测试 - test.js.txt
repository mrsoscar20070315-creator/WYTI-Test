  onLoad(options) {
    // 从首页进入（包括"再测一次"），重置所有进度
    if (!options.fromResult) {
      // 清除所有测试相关缓存
      wx.removeStorageSync('testScores')
      wx.removeStorageSync('currentQuestion')
      wx.removeStorageSync('testResult')
      // 重置为初始状态
      this.setData({
        currentQuestion: 0,
        scores: { M: 0, D: 0, P: 0, S: 0, E: 0, V: 0 },
        progress: Math.round((1 / this.data.questions.length) * 100)
      })
    } else {
      // 从结果页返回（如果需要保留进度的场景）
      const savedScores = wx.getStorageSync('testScores') || { M: 0, D: 0, P: 0, S: 0, E: 0, V: 0 }
      const savedQuestion = wx.getStorageSync('currentQuestion') || 0
      this.setData({
        scores: savedScores,
        currentQuestion: savedQuestion,
        progress: Math.round((savedQuestion + 1) / this.data.questions.length * 100)
      })
    }

    // 随机打乱每道题的选项顺序
    this.shuffleAllOptions()
  },

  // Fisher-Yates 洗牌算法
  shuffleArray(array) {
    const newArray = [...array]
    for (let i = newArray.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1))
      ;[newArray[i], newArray[j]] = [newArray[j], newArray[i]]
    }
    return newArray
  },

  // 随机打乱所有题目的选项
  shuffleAllOptions() {
    const shuffledQuestions = this.data.questions.map(question => ({
      ...question,
      options: this.shuffleArray(question.options)
    }))
    this.setData({
      questions: shuffledQuestions
    })
  },

  onShow() {
    // 页面显示时，强制确保状态正确（防止手机端缓存问题）
    const savedQuestion = wx.getStorageSync('currentQuestion')
    if (savedQuestion === null || savedQuestion === undefined || savedQuestion >= this.data.questions.length) {
      // 如果缓存无效或越界，重置为初始状态
      this.setData({
        currentQuestion: 0,
        scores: { M: 0, D: 0, P: 0, S: 0, E: 0, V: 0 },
        progress: Math.round((1 / this.data.questions.length) * 100)
      })
      wx.removeStorageSync('testScores')
      wx.removeStorageSync('currentQuestion')
      wx.removeStorageSync('testResult')
    }
  },

  // 选择选项
  handleOptionTap(e) {
    const { scores: optionScores } = e.currentTarget.dataset
    const { currentQuestion, questions, scores } = this.data

    // 累加分数
    const newScores = { ...scores }
    for (let key in optionScores) {
      newScores[key] = (newScores[key] || 0) + optionScores[key]
    }

    // 保存分数和当前题号
    wx.setStorageSync('testScores', newScores)
    wx.setStorageSync('currentQuestion', currentQuestion + 1)

    // 检查是否是最后一题
    if (currentQuestion >= questions.length - 1) {
      // 计算结果并跳转
      this.calculateAndNavigate(newScores)
    } else {
      // 下一题
      this.setData({
        currentQuestion: currentQuestion + 1,
        scores: newScores,
        progress: Math.round(((currentQuestion + 2) / questions.length) * 100)
      })
    }
  },

  // 计算结果并跳转
  calculateAndNavigate(scores) {
    const { departmentWeights, departmentDescriptions, departmentGrowth, crossDirections } = this.data
    const dimensions = ['M', 'D', 'P', 'S', 'E', 'V']

    // 归一化分数
    const total = Object.values(scores).reduce((a, b) => a + b, 0)
    const normalizedScores = {}
    dimensions.forEach(dim => {
      normalizedScores[dim] = total > 0 ? scores[dim] / total : 0
    })

    // 计算每个专业的匹配度
    const deptMatches = []
    for (let dept in departmentWeights) {
      let matchScore = 0
      const weights = departmentWeights[dept]
      for (let dim in weights) {
        matchScore += (normalizedScores[dim] || 0) * weights[dim]
      }
      deptMatches.push({ name: dept, score: matchScore })
    }

    // 按匹配度降序排序
    deptMatches.sort((a, b) => b.score - a.score)

    const top1 = deptMatches[0]
    const top2 = deptMatches[1]
    const diff = top1.score - top2.score
    const relativeDiff = top1.score > 0 ? diff / top1.score : 1

    let resultType, resultName, resultDesc, resultGrowth
    let matchedDepts = []

    // 判断是否需要推荐交叉方向（Top 2差距<12%且存在预定义的交叉方向）
    if (relativeDiff < 0.12) {
      const crossKey = this.getCrossKey(top1.name, top2.name)
      const crossDir = crossDirections[crossKey]
      if (crossDir) {
        resultType = 'cross'
        resultName = crossDir.name
        resultDesc = crossDir.description
        resultGrowth = crossDir.growth
        matchedDepts = [top1.name, top2.name]
      }
    }

    // 若不符合交叉条件，推荐单一专业（取Top 1）
    if (!resultType) {
      resultType = 'single'
      resultName = top1.name
      resultDesc = departmentDescriptions[top1.name]
      resultGrowth = departmentGrowth[top1.name]
      matchedDepts = [top1.name]
    }

    // 找出最低的能力维度
    let minDim = dimensions[0]
    let minScore = normalizedScores[dimensions[0]]
    dimensions.forEach(dim => {
      if (normalizedScores[dim] < minScore) {
        minScore = normalizedScores[dim]
        minDim = dim
      }
    })

    // 根据最低维度推荐补充成长建议
    const dimGrowthMap = {
      'M': '尝试参加机械创新、机器人等动手实践项目，培养实验操作和空间想象能力。',
      'D': '加强编程训练，学习Python或C++，尝试用代码解决实际问题，参与开源项目。',
      'P': '夯实数理基础，重视物理和数学课程，多读科普和学术文献，培养物理直觉。',
      'S': '关注系统层面的问题，学习用全局视角分析复杂工程，阅读工业工程或管理学入门书籍。',
      'E': '了解电路基础知识，尝试电子设计竞赛或Arduino项目，培养对电子系统的兴趣。',
      'V': '多关注国家重大需求和工程技术背后的社会价值，参与能源环保相关的社会实践。'
    }
    const supplementaryAdvice = dimGrowthMap[minDim]

    // 保存结果
    const resultData = {
      resultType,
      resultName,
      resultDesc,
      resultGrowth,
      matchedDepts,
      topMatches: deptMatches.slice(0, 5),
      normalizedScores,
      minDim,
      minScore,
      supplementaryAdvice
    }
    wx.setStorageSync('testResult', resultData)

    // 跳转到结果页
    wx.redirectTo({
      url: '/pages/result/result'
    })
  },

  // 生成交叉方向Key（排序后拼接，保证AB和BA得到同一Key）
  getCrossKey(dept1, dept2) {
    const sorted = [dept1, dept2].sort()
    return sorted[0] + '-' + sorted[1]
  }
})
